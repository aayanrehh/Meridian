INFRAN8N
1.1Create a new directory: mkdir meridian-n8n && cd meridian-n8n
1.2Create .env file in this directory with all variables from the env table in Step S.02. Never commit this file. Add .env to .gitignore immediately.
1.3Create docker-compose.yml with the exact content below. The n8n_data volume persists credentials across container restarts — without this, every restart destroys your LinkedIn OAuth token.
YAMLmeridian-n8n/docker-compose.yml
version: '3.8'
services:
  n8n:
    image: n8nio/n8n:latest
    restart: always
    ports:
      - "5678:5678"
    environment:
      # Public URL — set to ngrok during dev, Railway URL after deploy
      - N8N_WEBHOOK_URL=${N8N_WEBHOOK_URL}
      - N8N_BASIC_AUTH_ACTIVE=true
      - N8N_BASIC_AUTH_USER=admin
      - N8N_BASIC_AUTH_PASSWORD=${N8N_BASIC_AUTH_PASSWORD}
      - GENERIC_TIMEZONE=America/New_York
      - TZ=America/New_York
      # CORS: allow Next.js to POST to webhooks from both local and Vercel
      - N8N_CORS_ORIGIN=https://your-app.vercel.app,http://localhost:3000
      # Disable telemetry/personalization prompts
      - N8N_DIAGNOSTICS_ENABLED=false
      - N8N_PERSONALIZATION_ENABLED=false
      # Expose these so n8n Code nodes can access them via $env.*
      - ANTHROPIC_API_KEY=${ANTHROPIC_API_KEY}
      - SUPABASE_URL=${SUPABASE_URL}
      - SUPABASE_SERVICE_ROLE_KEY=${SUPABASE_SERVICE_ROLE_KEY}
      - SERPER_API_KEY=${SERPER_API_KEY}
    volumes:
      - n8n_data:/home/node/.n8n
volumes:
  n8n_data:
1.4Start n8n locally: docker compose up -d. Confirm it's running at http://localhost:5678.
1.5Start ngrok tunnel for local dev: ngrok http 5678. Copy the https://xxxx.ngrok-free.app URL. Set this as N8N_WEBHOOK_URL in your .env and restart: docker compose down && docker compose up -d.
⚠n8n must know its own public URL at startup — otherwise webhook URLs it generates will be wrong (pointing to localhost). If you change ngrok URLs, you must restart the container.

S.03
Create Supabase project + run full schema
DATABASE
3.1Create new project at supabase.com. Region: US East (closest to Railway). Save the password — you'll need it.
3.2Go to SQL Editor → New Query. Paste and run the entire schema below. This creates all 4 tables the architecture requires: clients, posts, deployments, scrape_items.
SQLSupabase SQL Editor → run once
-- ─── CLIENTS TABLE ──────────────────────────────────────────────
-- Multi-tenant: each row = one Meridian client (e.g., Jerah Reeves)
create table clients (
  id            uuid        default gen_random_uuid() primary key,
  created_at    timestamptz default now(),
  name          text        not null,
  title         text,                           -- "Gen-Z Business Dean"
  linkedin_urn  text,                           -- "urn:li:person:xxxx"
  voice_profile text,                           -- system prompt / bio blob
  core_thesis   text,                           -- "I help Gen-Z thrive..."
  followers     integer     default 0,
  auth_user_id  uuid        references auth.users(id) on delete cascade
);

-- ─── SCRAPE_ITEMS TABLE ─────────────────────────────────────────
-- Raw content ingested from Google News, Twitter/X, LinkedIn, Reddit, RSS
create table scrape_items (
  id          uuid        default gen_random_uuid() primary key,
  created_at  timestamptz default now(),
  client_id   uuid        references clients(id) on delete cascade,
  source      text        not null,   -- 'google_news' | 'twitter' | 'reddit' | 'rss'
  url         text,
  title       text,
  body        text,                    -- raw scraped text
  relevance   numeric,                -- 0.0–1.0, set by Claude
  used        boolean     default false  -- true after a draft references it
);

-- ─── POSTS TABLE ────────────────────────────────────────────────
-- AI-generated drafts before they enter the deployment queue
create table posts (
  id            uuid        default gen_random_uuid() primary key,
  created_at    timestamptz default now(),
  client_id     uuid        references clients(id) on delete cascade,
  asset_type    text        not null,   -- 'LinkedIn Post' | 'Comment Reply' | 'Carousel'
  focus         text        not null,   -- topic / context string shown in queue table
  content       text        not null,   -- the draft text
  scrape_item_id uuid       references scrape_items(id),  -- source material
  generation    integer     default 1,  -- increments on regenerate
  status        text        default 'draft'  -- 'draft' | 'queued' | 'approved'
);

-- ─── DEPLOYMENTS TABLE ──────────────────────────────────────────
-- Tracks every LinkedIn publish action — the Deployment Queue in the UI
create table deployments (
  id               uuid        default gen_random_uuid() primary key,
  created_at       timestamptz default now(),
  client_id        uuid        references clients(id) on delete cascade,
  post_id          uuid        references posts(id),
  content_snapshot text        not null,   -- snapshot of text at time of deploy
  status           text        default 'pending', -- 'pending'|'deployed'|'failed'|'scheduled'
  linkedin_post_id text,                       -- returned by LinkedIn API after publish
  scheduled_for    timestamptz,               -- null = deploy immediately
  deployed_at      timestamptz,
  error_message    text                        -- filled if status = 'failed'
);

-- ─── INDEXES ────────────────────────────────────────────────────
create index on deployments (client_id, status, created_at desc);
create index on posts (client_id, status, created_at desc);
create index on scrape_items (client_id, used, created_at desc);

-- ─── ROW LEVEL SECURITY ─────────────────────────────────────────
alter table clients     enable row level security;
alter table posts        enable row level security;
alter table deployments  enable row level security;
alter table scrape_items enable row level security;

-- Service role (n8n) bypasses RLS — it has full access via service_role key
-- Auth users can only see their own client's data — Phase 2 adds these policies

TypeScriptcomponents/deployment-sheet.tsx
// At top of component — add state
const [isDeploying, setIsDeploying] = useState(false);
const textareaRef = useRef<HTMLTextAreaElement>(null);

// Replace current onClick on "Approve & Deploy"
const handleApproveAndDeploy = async () => {
  const postText = textareaRef.current?.value?.trim();
  if (!postText) { toast.error("Post content is empty."); return; }
  setIsDeploying(true);
  try {
    const res = await fetch(process.env.NEXT_PUBLIC_N8N_WEBHOOK_DEPLOY!, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ text: postText, assetId: asset.id }),
    });
    const data = await res.json();
    if (res.ok && data.success) {
      toast.success("Post deployed to LinkedIn.", { description: "Live on Jerah's profile.", duration: 5000 });
      onClose();       // close the sheet
      refreshQueue(); // re-fetch Supabase deployments table
    } else {
      toast.error(data.message ?? "Deploy failed — check n8n execution log.");
    }
  } catch {
    toast.error("Cannot reach deploy service.", { description: "Is n8n running on Railway?" });
  } finally {
    setIsDeploying(false);
  }
};

// The button — swap your static one
<Button onClick={handleApproveAndDeploy} disabled={isDeploying} className="w-full">
  {isDeploying
    ? <><Loader2 className="mr-2 h-4 w-4 animate-spin" />Deploying...</>
    : "Approve & Deploy"}
</Button>

SQLSupabase SQL Editor → run after schema
-- Users can only read clients they own
create policy "client owner access" on clients
  for all using (auth_user_id = auth.uid());

-- Cascading: users access posts/deployments/scrape_items via client ownership
create policy "posts via client" on posts
  for all using (
    client_id in (select id from clients where auth_user_id = auth.uid())
  );

create policy "deployments via client" on deployments
  for all using (
    client_id in (select id from clients where auth_user_id = auth.uid())
  );

create policy "scrape_items via client" on scrape_items
  for all using (
    client_id in (select id from clients where auth_user_id = auth.uid())
  );
S.09
Wire Deployment Queue — poll Supabase from Next.js

9.1Create app/api/queue/route.ts — a server-side API route that queries Supabase deployments table for the authenticated user's pending items.
TypeScriptapp/api/queue/route.ts
import { createClient } from '@/lib/supabase/server'
import { NextResponse } from 'next/server'

export async function GET() {
  const supabase = createClient()
  const { data, error } = await supabase
    .from('deployments')
    .select('*, posts(asset_type, focus, content)')
    .eq('status', 'pending')
    .order('created_at', { ascending: false })
    .limit(20)

  if (error) return NextResponse.json({ error: error.message }, { status: 500 })
  return NextResponse.json({ queue: data })
}

const items = $input.all().map(i => i.json);
const scored = [];

for (const item of items) {
  scored.push({
    json: {
      ...item,
      prompt: `Rate the relevance of this content (0.0-1.0) to these topics:
- Gen-Z professional development and career transitions
- University business education and leadership
- Corporate culture, workforce trends, academic rigor

Content title: ${item.title}
Content snippet: ${item.body?.substring(0, 300)}

Respond with ONLY a JSON object: {"score": 0.0, "reason": "one sentence"}`
    }
  });
}
return scored;
12.6Add an HTTP Request node calling Claude API for each item's relevance prompt. Use the same Anthropic API call pattern as WF2. Parse the score from the response.
